063  /**
064   * loads system properties when configuration file of the name
065   * {@link #CHIMERA_SYSTEM_PROPERTIES_FILE} is found.
066   */
067  private static void loadChimeraSystemProperties() {
068    try {
069      InputStream is = Thread.currentThread().getContextClassLoader()
070          .getResourceAsStream(CHIMERA_SYSTEM_PROPERTIES_FILE);
071
072      if (is == null)
073        return; // no configuration file is found
074
075      // Load property file
076      Properties props = new Properties();
077      props.load(is);
078      is.close();
079      Enumeration<?> names = props.propertyNames();
080      while (names.hasMoreElements()) {
081        String name = (String) names.nextElement();
082        if (name.startsWith("chimera.")) {
083          if (System.getProperty(name) == null) {
084            System.setProperty(name, props.getProperty(name));
085          }
086        }
087      }
088    } catch (Throwable ex) {
089      System.err.println("Could not load '"
090          + CHIMERA_SYSTEM_PROPERTIES_FILE + "' from classpath: "
091          + ex.toString());
092    }
093  }

095  /**
096   * Forcibly free the direct buffer.
097   *
098   * @param buffer the bytebuffer to be freed.
099   */
100  public static void freeDirectBuffer(ByteBuffer buffer) {
101    if (buffer instanceof sun.nio.ch.DirectBuffer) {
102      final sun.misc.Cleaner bufferCleaner =
103          ((sun.nio.ch.DirectBuffer) buffer).cleaner();
104      bufferCleaner.clean();
105    }
106  }

108  /**
109   * Reads crypto buffer size.
110   *
111   * @param props The <code>Properties</code> class represents a set of
112   *              properties.
113   * @return the buffer size.
114   * */
115  public static int getBufferSize(Properties props) {
116    String bufferSizeStr = props.getProperty(
117        CHIMERA_CRYPTO_STREAM_BUFFER_SIZE_KEY);
118    if (bufferSizeStr == null || bufferSizeStr.isEmpty()) {
119      bufferSizeStr = System
120        .getProperty(CHIMERA_CRYPTO_STREAM_BUFFER_SIZE_KEY);
121    }
122    if (bufferSizeStr == null || bufferSizeStr.isEmpty()) {
123      return CHIMERA_CRYPTO_STREAM_BUFFER_SIZE_DEFAULT;
124    } else {
125      return Integer.parseInt(bufferSizeStr);
126    }
127  }

213  /**
214   * Checks and floors buffer size.
215   *
216   * @param cipher the {@link org.apache.commons.crypto.cipher.Cipher} instance.
217   * @param bufferSize the buffer size.
218   * @return the remaining buffer size.
219   */
220  public static int checkBufferSize(Cipher cipher, int bufferSize) {
221    checkArgument(bufferSize >= MIN_BUFFER_SIZE,
222        "Minimum value of buffer size is " + MIN_BUFFER_SIZE + ".");
223    return bufferSize - bufferSize % cipher.getTransformation()
224        .getAlgorithmBlockSize();
225  }

227  /**
228   * This method is only for Counter (CTR) mode. Generally the Cipher calculates the
229   * IV and maintain encryption context internally.For example a
230   * {@link javax.crypto.Cipher} will maintain its encryption context internally
231   * when we do encryption/decryption using the Cipher#update interface.
232   * <p/>
233   * Encryption/Decryption is not always on the entire file. For example,
234   * in Hadoop, a node may only decrypt a portion of a file (i.e. a split).
235   * In these situations, the counter is derived from the file position.
236   * <p/>
237   * The IV can be calculated by combining the initial IV and the counter with
238   * a lossless operation (concatenation, addition, or XOR).
239   * @see http://en.wikipedia.org/wiki/Block_cipher_mode_of_operation#Counter_.28CTR.29
240   *
241   * @param initIV initial IV
242   * @param counter counter for input stream position
243   * @param IV the IV for input stream position
244   */
245  public static void calculateIV(byte[] initIV, long counter, byte[] IV) {
246    checkArgument(initIV.length == AES_BLOCK_SIZE);
247    checkArgument(IV.length == AES_BLOCK_SIZE);
248
249    int i = IV.length; // IV length
250    int j = 0; // counter bytes index
251    int sum = 0;
252    while (i-- > 0) {
253      // (sum >>> Byte.SIZE) is the carry for addition
254      sum = (initIV[i] & 0xff) + (sum >>> Byte.SIZE);
255      if (j++ < 8) { // Big-endian, and long is 8 bytes length
256        sum += (byte) counter & 0xff;
257        counter >>>= 8;
258      }
259      IV[i] = (byte) sum;
260    }
261  }

263  /**
264   * Helper method to create a Cipher instance and throws only IOException.
265   *
266   * @param props The <code>Properties</code> class represents a set of
267   *              properties.
268   * @param transformation the CipherTransformation instance.
269   * @return the Cipher instance.
270   * @throws IOException if an I/O error occurs.
271   */
272  public static Cipher getCipherInstance(CipherTransformation transformation,
273      Properties props) throws IOException {
274    try {
275      return CipherFactory.getInstance(transformation, props);
276    } catch (GeneralSecurityException e) {
277      throw new IOException(e);
278    }
279  }

298   * @param errorMessage the exception message to use if the check fails;

300   *                     .valueOf(Object)</code>.

338  /**
339   * Splits class names sequence into substrings, Trim each substring into an
340   * entry,and returns an list of the entries.
341   *
342   * @param clazzNames a string consist of a list of the entries joined by a
343   *                   delimiter.
344   * @param separator  a delimiter for the input string.
345   * @return a list of class entries.
346   */
347  public static List<String> splitClassNames(String clazzNames,
348      String separator) {
349    List<String> res = new ArrayList<String>();
350    if (clazzNames == null || clazzNames.isEmpty()) {
351      return res;
352    }
353
354    for (String clazzName : clazzNames.split(separator)) {
355      clazzName = clazzName.trim();
356      if (!clazzName.isEmpty()) {
357        res.add(clazzName);
358      }
359    }
360    return res;
361  }
