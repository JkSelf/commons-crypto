018package org.apache.commons.crypto.utils;
019
020import java.lang.ref.WeakReference;
021import java.lang.reflect.Constructor;
022import java.util.Collections;
023import java.util.Map;
024import java.util.WeakHashMap;
025
026import org.apache.commons.crypto.cipher.Cipher;
027
028/**
029 * General utility methods for working with reflection.
030 */
031public class ReflectionUtils {
032
033  private static final Map<ClassLoader, Map<String, WeakReference<Class<?>>>>
034    CACHE_CLASSES = new WeakHashMap<ClassLoader, Map<String, WeakReference<Class<?>>>>();
035
036  private static ClassLoader classLoader;
037  static {
038    classLoader = Thread.currentThread().getContextClassLoader();
039    if (classLoader == null) {
040      classLoader = Cipher.class.getClassLoader();
041    }
042  }
043
044  /**
045   * Sentinel value to store negative cache results in {@link #CACHE_CLASSES}.
046   */
047  private static final Class<?> NEGATIVE_CACHE_SENTINEL =
048    NegativeCacheSentinel.class;
049
050  /**
051   * A unique class which is used as a sentinel value in the caching
052   * for getClassByName. {@link Cipher#getClassByNameOrNull(String)}.
053   */
054  private static abstract class NegativeCacheSentinel {}
055
056  /**
057   * Uses the constructor represented by this {@code Constructor} object to
058   * create and initialize a new instance of the constructor's
059   * declaring class, with the specified initialization parameters.
060   *
061   * @param klass the Class object.
062   * @param args array of objects to be passed as arguments to
063   *             the constructor call.
064   * @return a new object created by calling the constructor
065   *            this object represents.
066   */
067  @SuppressWarnings("rawtypes")
068  public static <T> T newInstance(Class<T> klass, Object ... args) {
069    try {
070      Constructor<T> ctor = null;
071
072      if (args.length == 0) {
073        ctor = klass.getDeclaredConstructor(new Class[] {});
074      } else {
075        Class[] argClses = new Class[args.length];
076        for (int i = 0; i < args.length; i++) {
077          argClses[i] = args[i].getClass();
078        }
079        ctor = klass.getDeclaredConstructor(argClses);
080      }
081      ctor.setAccessible(true);
082      return ctor.newInstance(args);
083    } catch (Exception e) {
084      throw new RuntimeException(e);
085    }
086  }
087
088  /** 
089   * Gets the value of the <code>name</code> property as a <code>Class</code>
090   * implementing the interface specified by <code>xface</code>.
091   * If no such property is specified, then <code>defaultValue</code> is 
092   * returned.An exception is thrown if the returned class does not
093   * implement the named interface.
094   * 
095   * @param name the class name of default implementation.
096   * @param defaultValue default value.
097   * @param xface the interface implemented by the named class.
098   * @return property value as a <code>Class</code>, 
099   *         or <code>defaultValue</code>.
100   */
101  public static <U> Class<? extends U> getClass(String name, 
102                                         Class<? extends U> defaultValue, 
103                                         Class<U> xface) {
104    try {
105      Class<?> theClass = null;
106      if (name != null && !name.isEmpty()) {
107        theClass = getClassByName(name);
108      }
109      if (theClass == null) {
110        theClass = defaultValue;
111      }
112      if (theClass != null && !xface.isAssignableFrom(theClass))
113        throw new RuntimeException(theClass+" not "+xface.getName());
114      else if (theClass != null)
115        return theClass.asSubclass(xface);
116      else
117        return null;
118    } catch (Exception e) {
119      throw new RuntimeException(e);
120    }
121  }
122
123  /** 
124   * Gets the value of the <code>name</code> property as a <code>Class</code>.
125   * If no such property is specified, then <code>defaultValue</code> is 
126   * returned.
127   * 
128   * @param name the class name.
129   * @param defaultValue default value.
130   * @return property value as a <code>Class</code>, 
131   *         or <code>defaultValue</code>. 
132   */
133  public static Class<?> getClass(String name, Class<?> defaultValue) {
134    String valueString = System.getProperty(name);
135    if (valueString == null)
136      return defaultValue;
137    try {
138      return getClassByName(valueString);
139    } catch (ClassNotFoundException e) {
140      throw new RuntimeException(e);
141    }
142  }
143
144  /**
145   * Loads a class by name.
146   * 
147   * @param name the class name.
148   * @return the class object.
149   * @throws ClassNotFoundException if the class is not found.
150   */
151  public static Class<?> getClassByName(String name)
152      throws ClassNotFoundException {
153    Class<?> ret = getClassByNameOrNull(name);
154    if (ret == null) {
155      throw new ClassNotFoundException("Class " + name + " not found");
156    }
157    return ret;
158  }
159  
160  /**
161   * Loads a class by name, returning null rather than throwing an exception
162   * if it couldn't be loaded. This is to avoid the overhead of creating
163   * an exception.
164   * 
165   * @param name the class name.
166   * @return the class object, or null if it could not be found.
167   */
168  private static Class<?> getClassByNameOrNull(String name) {
169    Map<String, WeakReference<Class<?>>> map;
170
171    synchronized (CACHE_CLASSES) {
172      map = CACHE_CLASSES.get(classLoader);
173      if (map == null) {
174        map = Collections.synchronizedMap(
175          new WeakHashMap<String, WeakReference<Class<?>>>());
176        CACHE_CLASSES.put(classLoader, map);
177      }
178    }
179
180    Class<?> clazz = null;
181    WeakReference<Class<?>> ref = map.get(name); 
182    if (ref != null) {
183       clazz = ref.get();
184    }
185     
186    if (clazz == null) {
187      try {
188        clazz = Class.forName(name, true, classLoader);
189      } catch (ClassNotFoundException e) {
190        // Leave a marker that the class isn't found
191        map.put(name, new WeakReference<Class<?>>(NEGATIVE_CACHE_SENTINEL));
192        return null;
193      }
194      // two putters can race here, but they'll put the same class
195      map.put(name, new WeakReference<Class<?>>(clazz));
196      return clazz;
197    } else if (clazz == NEGATIVE_CACHE_SENTINEL) {
198      return null; // not found
199    } else {
200      // cache hit
201      return clazz;
202    }
203  }
