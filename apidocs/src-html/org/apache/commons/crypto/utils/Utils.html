066  /**
067   * loads system properties when configuration file of the name
068   * {@link #COMMONS_CRYPTO_SYSTEM_PROPERTIES_FILE} is found.
069   */
070  private static void loadSystemProperties() {
071    try {
072      InputStream is = Thread.currentThread().getContextClassLoader()
073          .getResourceAsStream(COMMONS_CRYPTO_SYSTEM_PROPERTIES_FILE);
074
075      if (is == null)
076        return; // no configuration file is found
077
078      // Load property file
079      Properties props = new Properties();
080      props.load(is);
081      is.close();
082      Enumeration<?> names = props.propertyNames();
083      while (names.hasMoreElements()) {
084        String name = (String) names.nextElement();
085        if (name.startsWith(ConfigurationKeys.CONF_PREFIX)) {
086          if (System.getProperty(name) == null) {
087            System.setProperty(name, props.getProperty(name));
088          }
089        }
090      }
091    } catch (Throwable ex) {
092      System.err.println("Could not load '"
093          + COMMONS_CRYPTO_SYSTEM_PROPERTIES_FILE + "' from classpath: "
094          + ex.toString());
095    }
096  }

098  /**
099   * Forcibly free the direct buffer.
100   *
101   * @param buffer the bytebuffer to be freed.
102   */
103  public static void freeDirectBuffer(ByteBuffer buffer) {
104    if (buffer instanceof sun.nio.ch.DirectBuffer) {
105      final sun.misc.Cleaner bufferCleaner =
106          ((sun.nio.ch.DirectBuffer) buffer).cleaner();
107      bufferCleaner.clean();
108    }
109  }

111  /**
112   * Reads crypto buffer size.
113   *
114   * @param props The <code>Properties</code> class represents a set of
115   *              properties.
116   * @return the buffer size.
117   * */
118  public static int getBufferSize(Properties props) {
119    String bufferSizeStr = props.getProperty(
120        COMMONS_CRYPTO_STREAM_BUFFER_SIZE_KEY);
121    if (bufferSizeStr == null || bufferSizeStr.isEmpty()) {
122      bufferSizeStr = System
123        .getProperty(COMMONS_CRYPTO_STREAM_BUFFER_SIZE_KEY);
124    }
125    if (bufferSizeStr == null || bufferSizeStr.isEmpty()) {
126      return COMMONS_CRYPTO_STREAM_BUFFER_SIZE_DEFAULT;
127    } else {
128      return Integer.parseInt(bufferSizeStr);
129    }
130  }

221  /**
222   * Checks and floors buffer size.
223   *
224   * @param cipher the {@link org.apache.commons.crypto.cipher.Cipher} instance.
225   * @param bufferSize the buffer size.
226   * @return the remaining buffer size.
227   */
228  public static int checkBufferSize(Cipher cipher, int bufferSize) {
229    checkArgument(bufferSize >= MIN_BUFFER_SIZE,
230        "Minimum value of buffer size is " + MIN_BUFFER_SIZE + ".");
231    return bufferSize - bufferSize % cipher.getTransformation()
232        .getAlgorithmBlockSize();
233  }

235  /**
236   * This method is only for Counter (CTR) mode. Generally the Cipher calculates the
237   * IV and maintain encryption context internally.For example a
238   * {@link javax.crypto.Cipher} will maintain its encryption context internally
239   * when we do encryption/decryption using the Cipher#update interface.
240   * <p/>
241   * Encryption/Decryption is not always on the entire file. For example,
242   * in Hadoop, a node may only decrypt a portion of a file (i.e. a split).
243   * In these situations, the counter is derived from the file position.
244   * <p/>
245   * The IV can be calculated by combining the initial IV and the counter with
246   * a lossless operation (concatenation, addition, or XOR).
247   * @see http://en.wikipedia.org/wiki/Block_cipher_mode_of_operation#Counter_.28CTR.29
248   *
249   * @param initIV initial IV
250   * @param counter counter for input stream position
251   * @param IV the IV for input stream position
252   */
253  public static void calculateIV(byte[] initIV, long counter, byte[] IV) {
254    checkArgument(initIV.length == AES_BLOCK_SIZE);
255    checkArgument(IV.length == AES_BLOCK_SIZE);
256
257    int i = IV.length; // IV length
258    int j = 0; // counter bytes index
259    int sum = 0;
260    while (i-- > 0) {
261      // (sum >>> Byte.SIZE) is the carry for addition
262      sum = (initIV[i] & 0xff) + (sum >>> Byte.SIZE);
263      if (j++ < 8) { // Big-endian, and long is 8 bytes length
264        sum += (byte) counter & 0xff;
265        counter >>>= 8;
266      }
267      IV[i] = (byte) sum;
268    }
269  }

271  /**
272   * Helper method to create a Cipher instance and throws only IOException.
273   *
274   * @param props The <code>Properties</code> class represents a set of
275   *              properties.
276   * @param transformation the CipherTransformation instance.
277   * @return the Cipher instance.
278   * @throws IOException if an I/O error occurs.
279   */
280  public static Cipher getCipherInstance(CipherTransformation transformation,
281      Properties props) throws IOException {
282    try {
283      return CipherFactory.getInstance(transformation, props);
284    } catch (GeneralSecurityException e) {
285      throw new IOException(e);
286    }
287  }

303   * Checks the truth of an expression.
304   *
305   * @param expression   a boolean expression.
306   * @param errorMessage the exception message to use if the check fails;
307   *                     will be converted to a string using <code>String
308   *                     .valueOf(Object)</code>.

346  /**
347   * Splits class names sequence into substrings, Trim each substring into an
348   * entry,and returns an list of the entries.
349   *
350   * @param clazzNames a string consist of a list of the entries joined by a
351   *                   delimiter.
352   * @param separator  a delimiter for the input string.
353   * @return a list of class entries.
354   */
355  public static List<String> splitClassNames(String clazzNames,
356      String separator) {
357    List<String> res = new ArrayList<String>();
358    if (clazzNames == null || clazzNames.isEmpty()) {
359      return res;
360    }
361
362    for (String clazzName : clazzNames.split(separator)) {
363      clazzName = clazzName.trim();
364      if (!clazzName.isEmpty()) {
365        res.add(clazzName);
366      }
367    }
368    return res;
369  }
